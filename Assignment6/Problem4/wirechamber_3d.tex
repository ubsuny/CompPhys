\documentclass[tikz,border=6pt]{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot} % <-- required

% choose a viewing angle: (elevation, azimuth)
\tdplotsetmaincoords{60}{120}

\begin{document}


\begin{tikzpicture}[tdplot_main_coords, scale=0.9]

% Define half-dimensions
\def\Lx{10}
\def\Ly{2}
\def\Lz{6}

% Draw bounding box of chamber
\draw[thick] (-\Lx/2,-\Ly/2,-\Lz/2) -- (\Lx/2,-\Ly/2,-\Lz/2) --
                          (\Lx/2,\Ly/2,-\Lz/2) -- (-\Lx/2,\Ly/2,-\Lz/2) -- cycle;

\draw[thick] (-\Lx/2,-\Ly/2,\Lz/2) -- (\Lx/2,-\Ly/2,\Lz/2) --
                          (\Lx/2,\Ly/2,\Lz/2) -- (-\Lx/2,\Ly/2,\Lz/2) -- cycle;

\draw[thick] (-\Lx/2,-\Ly/2,-\Lz/2) -- (-\Lx/2,-\Ly/2,\Lz/2);
\draw[thick] (\Lx/2,-\Ly/2,-\Lz/2) -- (\Lx/2,-\Ly/2,\Lz/2);
\draw[thick] (\Lx/2,\Ly/2,-\Lz/2) -- (\Lx/2,\Ly/2,\Lz/2);
\draw[thick] (-\Lx/2,\Ly/2,-\Lz/2) -- (-\Lx/2,\Ly/2,\Lz/2);

% Wire x-positions
\foreach \x in {-4,-2,0,2,4} {
    % Draw vertical (z-directed) wires
    \draw[line width=1pt, blue] (\x,0,-\Lz/2) -- (\x,0,\Lz/2);
    % Draw a point at the center intersection
    \shade[ball color=blue] (\x,0,0) circle (0.15);
}

% Axis labels
\draw[->] (0,0,0) -- (6,0,0) node[right] {$x$};
\draw[->] (0,0,0) -- (0,3,0) node[above] {$y$};
\draw[->] (0,0,0) -- (0,0,4) node[above] {$z$};

\end{tikzpicture}

\end{document}
